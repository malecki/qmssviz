fit <- lm(dat$Death~dat$MoneyRaised)
ggplotRegression <- function (fit,col_group=1) {
require(ggplot2)
ggplot(fit$model, aes_string(x = names(fit$model)[2], y = names(fit$model)[1])) + 
  geom_point(color=col_group) +
  stat_smooth(method = "lm", col = "red") 
}
