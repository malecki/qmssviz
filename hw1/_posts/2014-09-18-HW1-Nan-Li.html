2.3D pie chart

library(plotrix)
opar <- par(no.readonly=TRUE)
par(mar=c(1,1,1,1), font=4)
pct <- round(d$Death / sum(d$Death)*100)
lbls <- paste(d$Name, " ", pct, "%", sep="")
pie3D(d$Death,labels=lbls, explode=0.1, labelcex=0.5, start=0, col=c("red","blue","brown","green","orange","yellow","pink","purple"), main="Death pie chart")

par(mar=c(1,1,1,1), font=4)

3. ggplot & annotation

DonationperDeath=(d$MoneyRaised / d$Death )* 1000
bp <- ggplot(d, aes(y=DonationperDeath,x=Death, color=Name, size=DonationperDeath)) +
  geom_point() +
  geom_text(aes(y=DonationperDeath+0.3, x=Death-50000, label=Name, colour=Name), size=5, hjust=0)
bp + theme(legend.position="none") +
  ggtitle("Donation per Death") + theme(plot.title = element_text(lineheight=.7, size=24, face="bold"))
